구조체 안에 배열 넣기

2014-12-01 17:08:13 | 비공개 | 준비중 

#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <string.h>
#include <sys/time.h>


struct ccc {
        int a;
        int b;
        int p[0];
};


struct ddd {
        int a;
        int *p;
        int b;
};

#define LENGTH 10000


int data[LENGTH];

int main(void)
{
        int i;

        struct ccc *s1;
        struct ddd *s2;

        struct timeval s, e;

        for (i = 0; i < LENGTH; i++)
                data[i] = (i*i + 256) % 1024;





        s1 = malloc(sizeof(struct ccc) + LENGTH * sizeof(long));

        printf("s1->%p p->%p\n", s1, s1->p);

        gettimeofday(&s, NULL);

        s1->a = 11;
        s1->b = 22;
        for (i = 0; i < LENGTH; i++)
                s1->p[i] = data[i];
        for (i = 0; i < LENGTH; i++)
                if (s1->p[i] == 0) {
                        printf("memory error\n");
                        break;
                }
        gettimeofday(&e, NULL);

        printf("%lu %lu\n", e.tv_sec - s.tv_sec, e.tv_usec - s.tv_usec);

        free(s1);





        s2 = malloc(sizeof(struct ddd) + LENGTH * sizeof(long));
        s2->p = (void *)(s2 + 1);

        printf("s2->p=%p\n", s2->p);

        gettimeofday(&s, NULL);

        s2->a = 11;
        s2->b = 22;
        for (i = 0; i < LENGTH; i++)
                s2->p[i] = data[i];
        for (i = 0; i < LENGTH; i++)
                if (s2->p[i] == 0) {
                        printf("memory error\n");
                        break;
                }
        gettimeofday(&e, NULL);

        printf("%lu %lu\n", e.tv_sec - s.tv_sec, e.tv_usec - s.tv_usec);

        free(s2);



        return 0;
}

.L3:
        ldr     r2, [r4, #4]!
        cmp     r4, ip
        str     r2, [r3, #4]!
        bne     .L3
        movw    r3, #10000
        b       .L6

r4가 data 배열이고, r3가 p입니다. p가 배열인것을 알기때문에 컴파일러는 !를 써서 자동으로 인덱싱을 해줍니다. 그래서 더 빠릅니다.

 


.L3:
        ldr     r2, [r3, lr]
        str     r2, [r1, r3]
        add     r3, r3, #4
        cmp     r3, ip
        bne     .L3
        mov     r3, #0
        movw    ip, #40000
        b       .L6

r1이 p이고 r3가 배열의 인덱스입니다. 컴파일러는 p만 봐서는 이게 배열에 대한 포인터인지 아니면 그냥 하나의 데이터에 대한 포인터인지 알 수 없습니다. 따라서 일일이 add 명령으로 인덱스를 계산해야합니다.
